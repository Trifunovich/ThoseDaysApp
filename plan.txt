name: menstrual-app-plan
version: 1.0.0
platform: continue
stack:
  backend: dotnet
  frontend: react + vite
  db: postgres
  pwa: true

models:
  reasoning: qwen3-coder:30b
  autocomplete: qwen2.5-coder:14b
  embed: nomic-embed-text:latest

tasks:
  - id: bootstrap
    title: Bootstrap dev environment
    steps:
      - run: docker-compose up -d
      - run: dotnet restore ./backend
      - run: npm install --prefix ./frontend
  - id: db-migrate
    title: Run DB migrations and seed
    steps:
      - run: dotnet ef database update --project backend/Api
      - run: dotnet run --project backend/Api --seed
  - id: start-dev
    title: Start dev servers
    steps:
      - run: dotnet watch run --project backend/Api
      - run: npm run dev --prefix frontend
  - id: generate-predictions
    title: Generate 15-cycle predictions
    steps:
      - run: dotnet run --project backend/Api --command generate-predictions --cycles 15
  - id: apply-corrections
    title: Apply user corrections and recalc
    steps:
      - run: dotnet run --project backend/Api --command apply-corrections
  - id: build
    title: Build for production
    steps:
      - run: dotnet publish backend/Api -c Release
      - run: npm run build --prefix frontend

endpoints:
  - GET /api/user/{id}/cycles
  - POST /api/user/{id}/cycles  # add period (startDate, duration)
  - PUT /api/user/{id}/cycles/{cycleId} # edit/correct
  - POST /api/user/{id}/predict?cycles=15
  - POST /api/user/{id}/toggle  # start/stop future painting
  - GET /api/user/{id}/stats

db_schema:
  tables:
    users:
      - id: uuid
      - email: text (nullable in phase1)
      - password_hash: text
      - is_active: bool
    cycles:
      - id: uuid
      - user_id: uuid
      - start_date: date
      - duration_days: int
      - created_at: timestamp
      - corrected: bool
    predictions:
      - id: uuid
      - user_id: uuid
      - predicted_start: date
      - predicted_duration: int
      - confidence: float

calculation:
  - anchor: "Cycle length = days between first day of cycle N and first day of cycle N+1"
  - default_cycle_length_days: 28
  - user_average: "mean of last N intervals (min 1). Use default 28 if only one cycle."
  - prediction_rule: "next_start = last_start + round(user_average)"
  - corrections: "When user edits a past cycle, recompute averages and regenerate next 15 predictions."

ui:
  main_screen: calendar view (month grid) + top status bar (avg length, avg interval)
  calendar_marks:
    - next_period: #ff6b6b (dark orange for immediate next)
    - subsequent_period: #ffb86b (soft orange)
    - later_periods: #ffd966 (pale yellow)
    - symbol: blood-drop icon (SVG)
  palette:
    primary: "#FF6B6B"
    accent: "#FFB86B"
    neutral: "#F7F7F8"
    text: "#1F2937"
  fonts:
    ui: "Inter"
    headings: "Merriweather"
  accessibility:
    - color_contrast: ensure WCAG AA for text on backgrounds
    - symbols + color (don’t rely on color alone)

security:
  - basic_auth: enabled (phase1)
  - require_unique_password: true (warn user to use different password)
  - data_retention: user can delete all data
  - disclaimer: show on first login and before enabling predictions

phase2:
  - add email login + optional notifications (no confirmation required per request)
  - analytics opt-in

notes:
  - UI should allow textual/compact/line views as alternate calendar renderers
  - store only minimal data (period start + duration)
